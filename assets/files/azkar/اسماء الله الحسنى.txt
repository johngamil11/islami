الرَّحْمٰنُ
كثير الرحمة، وهي صفة دالة على سعة رحمة الله بعباده.
#
الرَّحِيمُ
دائم الرحمة، وهو خاص بالمؤمنين يوم القيامة.
#
المَلِكُ
المتصرف المطلق في الكون، مالك كل شيء.
#
القُدُّوسُ
المنزه عن كل نقص وعيب، وهو الكامل في ذاته وصفاته.
#
السَّلَامُ
مصدر السلامة والأمان، وهو الذي سلم من كل عيب ونقص.
#
المُؤْمِنُ
الذي يصدق عباده ويمنحهم الأمن.
#
المُهَيْمِنُ
الشهيد والرقيب على كل شيء، المسيطر بقدرته.
#
العَزِيز
القوي الذي لا يُغلب، ذو العزة والقوة.
#
الجَبَّارُ
الذي يجبر الضعيف ويصلح الأمور.
#
المُتَكَبِّرُ
المتعالي عن صفات المخلوقين، الذي له الكبرياء والعظمة.
#
الخَالِقُ
الموجد للأشياء من العدم بقدرته وحكمته.
#
البَارِئُ
الذي خلق الخلق متناسبًا بلا خلل.
#
المُصَوِّر
الذي أعطى كل مخلوق صورته وشكله المميز.
#
الغَفَّار
كثير المغفرة لعباده مهما كثرت ذنوبهم.
#
القَهَّارُ
الغالب الذي لا يُقهر، وهو المسيطر على كل شيء.
#
الوَهَّابُ
كثير العطاء بلا مقابل.
#
الرَّزَّاقُ
الذي يرزق جميع خلقه بلا نقصان.
#
الفَتَّاحُ
الذي يفتح لعباده أبواب الخير.
#
العَلِيمُ
الذي يعلم كل شيء، الظاهر والباطن.
#
القَابِضُ
الذي يقبض الأرزاق بحكمته.
#
البَاسِطُ
الذي يبسط الرزق لعباده.
#
الخَافِضُ
الذي يخفض من يشاء من عباده.
#
الرَّافِعُ
الذي يرفع من يشاء بفضله.
#
المُعِزُّ
الذي يمنح العزة لمن يشاء.
#
المُذِلُّ
الذي يذل الظالمين والمتكبرين.
#
السَّمِيعُ
الذي يسمع كل شيء، سرًا وجهراً.
#
البَصِيرُ
الذي يرى كل شيء، ظاهرًا وباطنًا.
#
الحَكَم
الذي يفصل بين عباده بالعدل.
#
العَدْلُ
الذي لا يظلم أحدًا، العادل في كل شيء.
#
اللَّطِيفُ
البر بعباده، الذي يوصل لهم الخير بطرق خفية.
#
الخَبِيرُ
العالم ببواطن الأمور وخفاياها.
#
الحَلِيمُ
الذي لا يعجل بالعقوبة رغم القدرة عليها.
#
العَظِيم
العظيم في ذاته وصفاته وأفعاله.
#
الغَفُورُ
الذي يغفر الذنوب ويتجاوز عنها.
#
الشَّكُورُ
الذي يثيب القليل من العمل بالكثير من الثواب.
#
العَلِيُّ
المرتفع بذاته وصفاته، ذو العلو المطلق.
#
الكَبِيرُ
الذي كل شيء دونه، فلا شيء أعظم منه.
#
الحَفِيظُ
الحافظ لكل شيء، والذي يحفظ عباده.
#
المُقِيتُ
الذي يمد خلقه بما يحتاجونه.
#
الحسِيبُ
الكافي لعباده، الذي يحاسبهم بعدله.
#
الجَلِيلُ
صاحب العظمة والجلال.
#
الكَرِيمُ
الجواد المعطي بلا حد.
#
الرَّقِيبُ
الذي يراقب كل شيء ولا يغيب عنه شيء.
#
المُجِيبُ
الذي يستجيب لدعاء عباده.
#
الوَاسِع
الذي وسعت رحمته وعلمه كل شيء.
#
الحَكِيمُ
الذي يضع الأمور في مواضعها بحكمة.
#
الوَدُود
المحب لعباده، والمحبوب منهم.
#
المَجِيدُ
ذو المجد والكرم الواسع.
#
البَاعِثُ
الذي يبعث الخلق يوم القيامة.
#
الشَّهِيدُ
الذي لا يغيب عنه شيء، وهو حاضر دائمًا.
#
الحَقُّ
الموجود الحق الذي لا شك فيه.
#
الوَكِيلُ
المتوكل عليه في كل الأمور.
#
القَوِيُّ
صاحب القوة المطلقة.
#
المَتِينُ
شديد القوة، الذي لا يضعف.
#
الوَلِيُّ
القريب من عباده، المحب لهم.
#
الحَمِيدُ
المستحق للحمد في كل حال.
#
المُحْصِي
الذي أحصى كل شيء بعلمه.
#
المُبْدِئُ
الذي بدأ الخلق من العدم.
#
المُعِيدُ
الذي يعيد الخلق بعد الموت.
#
المُحْيِي
الذي يحيي الموتى.
#
المُمِيتُ
الذي يميت الأحياء بقدرته.
#
الحَيُّ
الذي له الحياة الكاملة المطلقة.
#
القَيُّوم
 القائم بنفسه والمقيم لغيره.
#
الصَّبُورُ
الذي لا يعجل بالعقوبة.
#
الأسماء الحسنى في القرآن

قال تعالى : "ولله الأسماء الحسنى فادعوه بها وذروا الذين يلحدون في أسمآئه سيجزون ما كانوا يعملون". [الأعراف - 180]
وقال تعالى : "قل ادعوا الله أو ادعوا الرحمن أيا ما تدعوا فله الأسماء الحسنى ولا تجهر بصلاتك ولا تخافت بها وابتغ بين ذلك سبيلا". [الإسراء - 110]
وقال تعالى : "الله لا إله إلا هو له الأسماء الحسنى". [طه - 8]
وقال تعالى : "هو الله الخالق البارئ المصور له الأسماء الحسنى يسبح له ما في السماوات والأرض وهو العزيز الحكيم". [الحشر - 24]
الأسماء الحسنى في الحديث

قال رسول الله صلى الله عليه وسلم : "إن لله تسعة وتسعين اسما مائة إلا واحدا من أحصاها دخل الجنة".
